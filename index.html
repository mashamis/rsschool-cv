    •     int FindMax(int A[], size_t size, size_t index)
    {
        int max = A[index], ind = index;
        for (size_t j = index; j < size; j++)
        {
            if (max < A[j])
            {
                max = A[j];
                ind = j;
            }
        }
        return ind;
    }
